package httphelpers

import (
	"encoding/json"
	"net/http"
)

func RespondErr(w http.ResponseWriter, httpStatus int, err error) {
	w.Header().Set("Content-Type", "application/json")
	w.WriteHeader(httpStatus)
	json.NewEncoder(w).Encode(map[string]any{
		"error": err.Error(),
	})
}

func RespondSuccess(w http.ResponseWriter, httpStatus int, resp any) {
	w.Header().Set("Content-Type", "application/json")
	w.WriteHeader(httpStatus)
	json.NewEncoder(w).Encode(resp)
}
