package delivery

import (
	"errors"
	httphelpers "gofronet-foundation/gofro-control/internal/utils/http_helpers"
	"net/http"

	"github.com/go-chi/chi/v5"
)

func (router *Router) Start(w http.ResponseWriter, r *http.Request) {
	nodeName := chi.URLParam(r, "node_name")
	if nodeName == "" {
		httphelpers.RespondErr(w, http.StatusBadRequest, errors.New("node_name required"))
		return
	}

	node, err := router.nodeManager.GetNodeByName(r.Context(), nodeName)
	if err != nil {
		httphelpers.RespondErr(w, http.StatusInternalServerError, err)
		return
	}

	if err := node.Start(r.Context()); err != nil {
		httphelpers.RespondErr(w, http.StatusInternalServerError, err)
		return
	}

	httphelpers.RespondSuccess(w, http.StatusOK, map[string]string{"status": "ok"})
}
